<script>
    let currentUser = null;
    let gameInterval = null;
    let currentMultiplier = 1.00;
    let isCrashed = false;
    let currentBet = 0;
    let nextCrashPoint = 0; // Global variable to store the pre-determined crash point

    function showSection(id) {
        document.querySelectorAll('.section, .container').forEach(s => s.classList.remove('active'));
        document.getElementById(id).classList.add('active');
        if(currentUser) {
            updateUI();
            document.getElementById('predictor-ui').style.display = 'block';
            updatePredictor(); // Generate first prediction on login
        }
    }

    function handleAuth(type) {
        const u = document.getElementById('username').value.trim();
        const p = document.getElementById('password').value.trim();
        if(!u || !p) return alert("Fill all fields");

        if(type === 'reg') {
            if(localStorage.getItem(u)) return alert("User exists");
            localStorage.setItem(u, JSON.stringify({ password: p, balance: 500, history: [] }));
            alert("Account Created! 500 KES Bonus added.");
        } else {
            const data = JSON.parse(localStorage.getItem(u));
            if(data && data.password === p) {
                currentUser = u;
                document.getElementById('main-nav').style.display = 'flex';
                document.getElementById('user-name').innerText = u;
                showSection('game-sec');
            } else {
                alert("Wrong credentials");
            }
        }
    }

    function updatePredictor() {
        const status = document.getElementById('pred-status');
        const val = document.getElementById('next-prediction');
        
        status.innerText = "CALCULATING...";
        status.style.color = "var(--gold)";
        val.innerText = "Scanning Server...";

        // Set the exact crash point for the NEXT round
        nextCrashPoint = parseFloat((1 + Math.random() * 4.5).toFixed(2));

        setTimeout(() => {
            status.innerText = "PREDICTION READY";
            status.style.color = "var(--primary)";
            val.innerText = "Next Crash @ " + nextCrashPoint.toFixed(2) + "x";
        }, 1500);
    }

    function startGame() {
        const data = JSON.parse(localStorage.getItem(currentUser));
        currentBet = parseFloat(document.getElementById('bet-amount').value);

        if(isNaN(currentBet) || currentBet < 10) return alert("Min bet is 10 KES");
        if(currentBet > data.balance) return alert("Insufficient Balance");
        if(nextCrashPoint === 0) return alert("Wait for AI Prediction");

        data.balance -= currentBet;
        localStorage.setItem(currentUser, JSON.stringify(data));
        
        isCrashed = false;
        currentMultiplier = 1.00;
        
        document.getElementById('start-btn').style.display = 'none';
        document.getElementById('cashout-btn').style.display = 'block';
        document.getElementById('crash-overlay').style.display = 'none';
        document.getElementById('multiplier-display').style.color = 'var(--primary)';
        document.getElementById('cloud-bg').style.display = 'block';
        document.getElementById('cloud-bg').style.animation = 'moveClouds 3s linear infinite';

        const plane = document.getElementById('plane');
        plane.style.display = 'block';

        gameInterval = setInterval(() => {
            currentMultiplier += 0.02; 
            document.getElementById('multiplier-display').innerText = currentMultiplier.toFixed(2) + "x";

            let x = Math.min((currentMultiplier - 1) * 80, 350); 
            let y = Math.max((currentMultiplier - 1) * -50, -200);
            plane.style.transform = `translate(${x}px, ${y}px)`;

            // Check if we hit the exact predicted crash point
            if(currentMultiplier >= nextCrashPoint) {
                crash();
            }
        }, 100);
        updateUI();
    }

    function crash() {
        clearInterval(gameInterval);
        isCrashed = true;
        document.getElementById('multiplier-display').innerText = "CRASHED";
        document.getElementById('multiplier-display').style.color = 'var(--danger)';
        document.getElementById('crash-overlay').style.display = 'flex';
        document.getElementById('start-btn').style.display = 'block';
        document.getElementById('cashout-btn').style.display = 'none';
        document.getElementById('plane').style.display = 'none';
        document.getElementById('cloud-bg').style.animation = 'none';
        
        const data = JSON.parse(localStorage.getItem(currentUser));
        data.history.unshift(`Lost KES ${currentBet} @ ${currentMultiplier.toFixed(2)}x`);
        localStorage.setItem(currentUser, JSON.stringify(data));
        
        // Refresh predictor with a NEW exact value for the next round
        updatePredictor();
        updateUI();
    }

    function cashOut() {
        if(isCrashed) return;
        clearInterval(gameInterval);
        
        const winnings = currentBet * currentMultiplier;
        const data = JSON.parse(localStorage.getItem(currentUser));
        data.balance += winnings;
        data.history.unshift(`WON KES ${winnings.toFixed(2)} @ ${currentMultiplier.toFixed(2)}x`);
        localStorage.setItem(currentUser, JSON.stringify(data));

        document.getElementById('multiplier-display').style.color = 'var(--gold)';
        document.getElementById('start-btn').style.display = 'block';
        document.getElementById('cashout-btn').style.display = 'none';
        document.getElementById('cloud-bg').style.animation = 'none';
        
        updatePredictor();
        updateUI();
    }

    function updateUI() {
        const data = JSON.parse(localStorage.getItem(currentUser));
        if(!data) return;
        document.getElementById('user-balance').innerText = data.balance.toFixed(2);
        document.getElementById('history-list').innerHTML = data.history.slice(0, 5)
            .map(h => `<div class="history-item">${h}</div>`).join('');
    }

    function simulateDeposit() {
        const amt = parseFloat(document.getElementById('dep-amount').value);
        if(amt < 50) return alert("Min deposit 50 KES");
        
        const data = JSON.parse(localStorage.getItem(currentUser));
        data.balance += amt;
        localStorage.setItem(currentUser, JSON.stringify(data));
        alert("Deposit Successful!");
        showSection('game-sec');
    }

    function handleWithdraw() {
        const amt = parseFloat(document.getElementById('withdraw-amount').value);
        const data = JSON.parse(localStorage.getItem(currentUser));
        if(amt > data.balance) return alert("Insufficient funds");
        
        data.balance -= amt;
        localStorage.setItem(currentUser, JSON.stringify(data));
        alert("Withdrawal Processed!");
        updateUI();
    }

    function logout() {
        currentUser = null;
        location.reload();
    }

    window.onload = () => {
        // Only run if user is already in session
        if(currentUser) updatePredictor();
    };
</script>
